Example 2.6.1.

\begin{align*} \lim_{t\to\infty}\frac{3t^2+5t}{3-5t^2}&=\lim_{t\to\infty}\left(\frac{3t^2+5t}{3-5t^2}\cdot\frac{\sfrac{1}{t^2}}{\sfrac{1}{t^2}}\right)\\ &=\lim_{t\to\infty}\frac{3+\sfrac{5}{t}}{\sfrac{3}{t^2}-5}&&\text{No longer indeterminate form}\\ &=\frac{\lim_{t\to\infty}\left(3+\sfrac{5}{t}\right)}{\lim_{t\to\infty}\left(\sfrac{3}{t^2}-5\right)}&&\knowl{./knowl/xref/lla5.html}{\text{Theorem A.0.8}}\\ &=\frac{\lim_{t\to\infty}3+\lim_{t\to\infty}\frac{5}{t}}{\lim_{t\to\infty}\frac{3}{t^2}-\lim_{t\to\infty}5}&&\knowl{./knowl/xref/lla1.html}{\text{Theorem A.0.4}}, \knowl{./knowl/xref/lla2.html}{\text{Theorem A.0.5}}\\ &=\frac{3+0}{0-5}&&\knowl{./knowl/xref/llr2.html}{\text{Theorem A.0.2}}, \knowl{./knowl/xref/llr3.html}{\text{Theorem A.0.3}}\\ &=-\frac{3}{5} \end{align*}
